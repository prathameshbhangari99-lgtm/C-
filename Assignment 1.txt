#include <iostream>
using namespace std;

struct Student
{
    int rn;
    string name;
    int cgpa;
};
void display(Student s[],int lmt)
{
    for(int i=0;i<lmt;i++)
    {
        cout<<"\nRoll No : "<<s[i].rn;
        cout<<"\nName : "<<s[i].name;
        cout<<"\nCGPA : "<<s[i].cgpa;
        cout<<endl;
    }
}
void insertionSort(Student s[],int lmt)
{
    for(int i=1;i<lmt;i++)
    {
        int j=i-1;
        Student temp=s[i];
        while(j>=0 && s[j].rn>temp.rn)
        {
            s[j+1]=s[j];
            j--;
        }
        s[j+1]=temp;
    }
    cout<<"\n\nStudent Record Sorted by Roll No : \n";
    display(s,lmt);
}
void shellSort(Student s[],int lmt)
{
    Student temp;
    int gap;
    for(int gap=lmt/2;gap>=1;gap=gap/2)
    {
        for(int j=gap;j<lmt;j++)
        {
            for(int i=j-gap;i>=0;i=i-gap)
            {
                if(s[i].name > s[i+gap].name)
                {
                    temp=s[i];
                    s[i]=s[i+gap];
                    s[i+gap]=temp;
                }
                else
                {
                    break;
                }
            }
        }
    }
    cout<<"\nStudent Record Sorted by Name : \n";
    display(s,lmt);
}
int getMaxVal(Student s[],int lmt)
{
    int mx=s[0].cgpa;
    for(int i=0;i<lmt;i++)
    {
        if(s[i].cgpa > mx)
        {
            mx=s[i].cgpa;
        }
    }
    return mx;
}
void countSort(Student s[],int lmt,int pos)
{
    int range=10;
    int count[range]={0};
    Student b[15];
    for(int i=0;i<lmt;i++)
    {
        ++count[(s[i].cgpa/pos)%10];
    }
    for(int i=1;i<range;i++)
    {
        count[i]=count[i]+count[i-1];
    }
    for(int i=lmt-1;i>=0;i--)
    {
        b[--count[(s[i].cgpa/pos)%10]]=s[i];
    }
    for(int i=0;i<lmt;i++)
    {
        s[i]=b[i];
    }
    
}

void radixSort(Student s[],int lmt)
{
    int maxVal = getMaxVal(s,lmt);
    for(int pos=1;maxVal/pos>0;pos=pos*10)
    {
        countSort(s,lmt,pos);
    }
    cout<<"\nStudent record sorted by CGPA : \n";
    display(s,lmt);
}
