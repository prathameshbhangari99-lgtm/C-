#include<iostream>
using namespace std;
struct Song
{
    string title;
    string artist;
    string duration;
    Song *next;
};
Song *temp, *pev;
Song *head=NULL;
void addSong()
{
    Song *newSong = new Song();
    newSong->next=NULL;
    cout<<"Enter Song Name : ";
    cin>>newSong->title;
    cout<<"Enter Song Artist : ";
    cin>>newSong->artist;
    cout<<"Enter Song Duration : ";
    cin>>newSong->duration;
    
    if(head==NULL)
    {
        head=newSong;
    }
    else
    {
        temp=head;
        while(temp->next!=NULL)
        {
            temp=temp->next;
        }
        temp->next=newSong;
    }
    cout<<"Song Added....\n";
}
void removeSong()
{
    string t;
    cout<<"\nEnter Title of the Song to delete : ";
    cin>>t;
    if(head==NULL)
    {
        cout<<"\nList is Emptyyy\n";
    }
    else
    {
        temp=head;
        pev=NULL;
        while(temp!=NULL && temp->title!=t)
        {
            pev=temp;
            temp=temp->next;
        }
        if(temp==NULL)      //last parenta gela ani sapdla nahi
        {
            cout<<"\nSong not found\n";
            return;
        }
        if(pev==NULL)      //start lach sapdla
        {
            head=temp->next;
            delete temp;
        }
        else{
            pev->next = temp->next;
            delete temp;
        }
        cout<<"\nSong deleted!!!\n";
    }
}
void display()
{
    if(head==NULL)
    {
        cout<<"\nList Is Empty";
        return;
    }
    temp=head;
    cout<<"\n******PLAYLIST******";
    while(temp != NULL)
    {
        cout<<endl<<temp->title<<" | "<<temp->artist<<" | "<<temp->duration<<" mins ";
        temp=temp->next;
    }
    cout<<"\n\n";
}
void playSong()
{
    string t;
    if(head==NULL)
    {
        cout<<"\nPlaylist Is Empty\n";
        return;
    }
    cout<<"\nEnter Song To Play : ";
    cin>>t;
    temp=head;
    while(temp!=NULL && temp->title!=t)
    {
        temp=temp->next;
    }
    if(temp==NULL)
    {
        cout<<"\nSong Not Found";
        return;
    }
    else{
        cout<<"\nNow Playing : "<<temp->title<<" by "<<temp->artist<<endl;
    }
}
int main()
{
    int ch;
    do{
        cout<<"\n-------MENU-------";
        cout<<"\n1.Add Song";
        cout<<"\n2.Remove Song";
        cout<<"\n3.Display Playlist";
        cout<<"\n4.Play a Song";
        cout<<"\n5.Exit";
        cout<<"\nEnter Your Choice : ";
        cin>>ch;
        switch(ch)
        {
            case 1:
            addSong();
            break;
            case 2:
            removeSong();
            break;
            case 3:
            display();
            break;
            case 4:
            playSong();
            break;
            case 5:
            cout<<"\nExiting...";
            exit(0);
            break;
            default:
            cout<<"\nINVALID CHOICE";
        }
    }while(ch!=5);
    return 0;
}
