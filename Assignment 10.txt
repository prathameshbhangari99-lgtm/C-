#include <iostream>
using namespace std;
struct Job
{
    int job_ID;
    int job_pr;
    Job *next;
    Job *prev;
};
Job *front=NULL;
Job *rear=NULL;
void addJobFront()
{
    int id,pr;
    Job *nn=new Job();
    cout<<"\nEnter Job ID : ";
    cin>>id;
    cout<<"\nEnter Job Priority : ";
    cin>>pr;
    
    nn->job_ID=id;
    nn->job_pr=pr;
    nn->next=front;
    nn->prev=NULL;
    
    if(front==NULL)
    {
        front=rear=nn;
    }
    else{
        front->prev=nn;
        front=nn;
    }
    cout<<"\nJob "<<nn->job_ID<<" added to the FRONT\n";
}

}

{
    if(front==NULL)
    {
        cout<<"\nNo Jobs To Remove \n";

        front->prev=NULL;
    }

}

}

{
    Job *temp=front;
