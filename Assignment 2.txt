#include <iostream>
using namespace std;
struct Employee
{
    int emp_ID;
    string emp_NAME;
    int emp_SALARY;
};
void swap(Employee &e1,Employee &e2)
{
    Employee temp;
    temp=e1;
    e1=e2;
    e2=temp;
}
int partition(Employee e[],int lb,int ub)
{
    int start=lb;
    int end=ub;
    int pivot = e[lb].emp_ID;
    while(start<end)
    {
        while(e[start].emp_ID <= pivot && start<ub)
        {
            start++;
        }
        while(e[end].emp_ID > pivot && end>lb)
        {
            end--;
        }
        if(start<end)
        {
            swap(e[start],e[end]);
        }
    }
    swap(e[end],e[lb]);
    return end;
    
}
void quickSort(Employee e[],int lb,int ub)
{
    if(lb<ub)
    {
        int loc=partition(e,lb,ub);
        quickSort(e,lb,loc-1);
        quickSort(e,loc+1,ub);
       
    }
}
void merge(Employee e[],int lb,int mid,int ub)
{
    int i=lb;
    int j=mid+1;
    int k=lb;
    Employee b[50];
    while(i<=mid && j<=ub)
    {
        if(e[i].emp_NAME < e[j].emp_NAME)
        {
            b[k]=e[i];
            i++;
        }
        else
        {
            b[k]=e[j];
            j++;
        }
        k++;
    }
    if(i>mid)
    {
        while(j<=ub)
        {
            b[k]=e[j];
            j++;
            k++;
        }
    }
    else
    {
        while(i<=mid)
        {
            b[k]=e[i];
            i++;
            k++;
        }
    }
    for(int k=lb;k<=ub;k++)
    {
        e[k]=b[k];
    }
}
void mergeSort(Employee e[],int lb,int ub)
{
    if(lb<ub)
    {
        int mid=(lb+ub)/2;
        mergeSort(e,lb,mid);
        mergeSort(e,mid+1,ub);
        merge(e,lb,mid,ub);
       
    }
}
void display(Employee e[],int n)
{
    for(int i=0;i<n;i++)
    {
         cout<<"\n\nID : "<<e[i].emp_ID;
        cout<<"\nNAME : "<<e[i].emp_NAME;
        cout<<"\nSALARY : "<<e[i].emp_SALARY;
    }
}
