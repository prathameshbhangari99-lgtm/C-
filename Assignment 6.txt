#include <iostream>
using namespace std;
struct Node{
    int coef;
    int expo;
    Node *next;
};
void insert(Node*& head,int coef,int expo)
{
    Node *nn=new Node();
    nn->coef=coef;
    nn->expo=expo;
    nn->next=NULL;
    if(head==NULL)
    {
        head=nn;
        return;
    }
    else{
        Node *temp=head;
        while(temp->next!=NULL)
        {
            temp=temp->next;   
        }
        temp->next=nn;
    }
}
Node *addPoly(Node *poly1, Node *poly2)
{
    Node *res=NULL;
    while(poly1!=NULL && poly2!=NULL)
    {
        if(poly1->expo == poly2->expo)
        {
            insert(res,poly1->coef+poly2->coef,poly1->expo);
            poly1=poly1->next;
            poly2=poly2->next;
        }
        else if(poly1->expo>poly2->expo)
        {
            insert(res,poly1->coef,poly1->expo);
            poly1=poly1->next;
        }
        else{
            insert(res,poly2->coef,poly2->expo);
            poly2=poly2->next;
        }
    }
    while(poly1!=NULL)
    {
        insert(res,poly1->coef,poly1->expo);
        poly1=poly1->next;
    }
    while(poly2!=NULL)
    {
        insert(res,poly2->coef,poly2->expo);
        poly2=poly2->next;
    }
    return res;
}
